\begin{enumerate}
% การหาค่าของฟังก์ชัน
\item \textbf{การหาค่าของฟังก์ชัน} 

\begin{enumerate}
\item กำหนด $f(x) = 2x+3$ จงหา
\begin{tasks}(4)
\task $f(0)$
\vspace{30pt}
\task $f(2)$
\task $f(-3)$
\task $f(a+b)$
\end{tasks}
\item กำหนด $g(x) = x^2+2x$ จงหา
\begin{tasks}(4)
\task $g(1)$
\vspace{30pt}
\task $g(-1)$
\task $g(x+1)$
\task $g\left(\frac{1}{\theta}\right)$
\end{tasks}
\item กำหนด $h(x) = \frac{x+3}{x-2}$ จงหา
\begin{tasks}(4)
\task $h(-2)$
\vspace{30pt}
\task $h(3)$
\task $h(7)$
\task $h(a+2)$
\end{tasks}
\end{enumerate}

% พีชคณิตของฟังก์ชัน
\item \textbf{พีชคณิตของฟังก์ชัน} 

\begin{enumerate}
\item กำหนดให้ $f(x) = 3x$ และ $g(x) = x+4$ จงหา
\begin{tasks}(3)
\task $2f(x)$
\vspace{50pt}
\task $f(x) + g(x)$
\task $f(x)g(x)$
\end{tasks}
\item กำหนดให้ $f(x) = x^2+2$ และ $g(x) = x^3-1$ จงหา
\begin{tasks}(3)
\task $-f(x)$
\vspace{50pt}
\task $f(x)+g(x)$
\task $f(x)g(x)$
\end{tasks}
\item กำหนดให้ $f(x) = x^2-1$ และ $g(x) = x-1$ จงหา
\begin{tasks}(3)
\task $xg(x)$
\vspace{50pt}
\task $f(x) + xg(x)$
\task $\frac{f(x)}{g(x)}$
\end{tasks}
\end{enumerate}

% ฟังก์ชันประกอบ
\item \textbf{ฟังก์ชันประกอบ} 

กำหนดฟังก์ชัน $f(x)$ และ $g(x)$ แล้วฟังก์ชันประกอบนิยามด้วย
\[ (f \circ g) (x) =  \rule{100pt}{.5pt} \]

กำหนด $f(x)$ และ $g(x)$ มาให้ในแต่ละข้อ จงหา $(f \circ g)(x)$ และ $(g \circ f)(x)$
\begin{tasks}(3)
\task $f(x) = 3x, g(x) = x^2$
\vspace{150pt}
\task $f(x) = x^3,  g(x) = 2x-1$
\task $f(x) = x^2+1, g(x) = x^2-1$
\end{tasks}
จงใช้ $(f \circ g)(x)$ และ $(g \circ f)(x)$ ที่หามาได้ข้างบน เพื่อหา $(f \circ g)(0)$ และ $(f \circ g)(1)$
\vspace{100pt}

% ฟังก์ชันผกผัน
\item \textbf{ฟังก์ชันผกผัน} 

กำหนดฟังก์ชัน $f(x)$ ที่เป็นฟังก์ชัน \rule{100pt}{.5pt} แล้วฟังก์ชันผกผัน $f^{-1}\,(x)$ มีสมบัติว่า
\[y = f(x) \then \rule{100pt}{.5pt}\]

กำหนดฟังก์ชัน $f(x)$ จงหาฟังก์ชันผกผัน $f^{-1}\,(x)$
\begin{tasks}(4)
\task $f(x) = 2x+3$
\vspace{100pt}
\task $f(x) = \sqrt{3x-1}$
\task $f(x) = x^3+4$
\task $f(x) = \frac{x+1}{x-2}$
\end{tasks}



\end{enumerate}
