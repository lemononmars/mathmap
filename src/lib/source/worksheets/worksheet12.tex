\begin{enumerate}
%1

\item \textbf{เทคนิคปริพันธ์แยกส่วน}

สูตรปริพันธ์แยกส่วนคือ

\[ \int u \, dv = \rule{200pt}{1pt}\]

จงหาปริพันธ์ต่อไปนี้ ด้วยเทคนิคการแยกส่วน
\begin{tasks}[after-skip=150pt, after-item-skip=150pt](2)
\task $\int  x \cos x \, dx$
\task $\int x \ln x \, dx$
\task $\int \frac{\ln x}{x^3} \, dx$
\task $\int (2+3x)e^{4x} \, dx$
\end{tasks}

\newpage

\item \textbf{การหาปริพันธ์ด้วยการแทนค่าฟังก์ชันตรีโกณมิติ}

\begin{itemize}
\item ถ้าปริพันธ์อยู่ในรูป $\sqrt{a^2 - b^2 x^2}$ หรือ $a^2 - b^2 x^2$ ให้แทนค่า
\[ x = \rule{100pt}{1pt} \]
\item ถ้าปริพันธ์อยู่ในรูป $\sqrt{a^2 + b^2 x^2}$ หรือ $a^2 + b^2 x^2$ ให้แทนค่า
\[ x = \rule{100pt}{1pt} \]
\item ถ้าปริพันธ์อยู่ในรูป $\sqrt{b^2 x^2 - a^2 }$ หรือ $ b^2 x^2 -  a$ ให้แทนค่า
\[ x = \rule{100pt}{1pt} \]
\end{itemize}

จงหาปริพันธ์ต่อไปนี้ ด้วยการแทนค่าฟังก์ชันตรีโกณมิติที่เหมาะสม
\begin{tasks}[after-item-skip = 150pt, after-skip=150pt](2)
\task $\int \sqrt{4 - 9x^2} \, dx$
\task $\int \frac{1}{x^2\sqrt{25-x^2}} \, dx$
\task $\int \frac{x^2}{\sqrt{9-x^2}} \, dx$
\task $\int \sqrt{25 + 16x^2} \, dx$
\end{tasks}


\end{enumerate}
