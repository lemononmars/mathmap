\begin{enumerate}
%1


\item \textbf{เทคนิคปริพันธ์ฟังก์ชันตรีโกณมิติ}

จงหาปริพันธ์ต่อไปนี้ โดยใช้เอกลักษณ์ตรีโกณมิติ

\begin{tasks}[after-skip=150pt, after-item-skip=150pt](3)
\task $\int \cos^2 x \, dx$
\task $\int \cos^2 x - \sin^2 x \,dx$
\task $\int \sin 3x \sin 2x \, dx$
\end{tasks}

%2
\item \textbf{เทคนิคปริพันธ์ฟังก์ชันตรีโกณมิติยกกำลัง}

จงหาปริพันธ์ต่อไปนี้ 
\begin{tasks}[after-skip=100pt, after-item-skip=100pt](2)
\task $\int \sin x \cos^4 x \, dx$ 
\task $\int \sin^{6} x \cos ^3 x \, dx$
\task $\int \frac{\sin^3 x}{\cos^8 x} \, dx$
\task $\int \cos^2 (4x) \, dx$
\end{tasks}

\newpage
\item \textbf{ปริพันธ์ฟังก์ชันไฮเปอร์โบลิก}

จงหาปริพันธ์ต่อไปนี้

\begin{tasks}[after-skip=80pt, after-item-skip=80pt](4)
\task $\int \sinh(1-2x) \, dx$
\task $\int \cosh(4x) \,dx$
\task $\int x\tanh(x^2) \, dx$
\task $\int \frac{1}{x^2} \coth\left(\frac{1}{x}\right) \, dx$
\end{tasks}


\item \textbf{ปริพันธ์ฟังก์ชันในรูป $u^2 \pm a^2$ และ $a^2 \pm u^2$}

จงหาปริพันธ์ต่อไปนี้
\begin{tasks}[after-skip=120pt, after-item-skip=120pt](2)
\task $\int \frac{1}{4 + x^2} \, dx$
\task $\int \frac{1}{16 - 9x^2} \, dx$
\task $\int \frac{1}{(2x+7)^2 - 25} \, dx$
\task $\int \frac{1}{x^2 + 2x + 5} \, dx$
\task $\int \frac{4}{\sqrt{2 - x^2}} \, dx$
\task $\int \frac{-2}{\sqrt{4x^2 + 36}} \, dx$
\end{tasks}
