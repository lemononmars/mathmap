\begin{enumerate}


%1
\item \textbf{พื้นที่ระหว่างเส้นโค้ง}

ถ้า $g(x) \geq f(x)$ สำหรับทุกค่า $x$ ในช่วง $[a,b]$ แล้วพื้นที่ระหว่างเส้นโค้ง $y=f(x)$, $y=g(x)$ เส้นตรง $x =a$ และเส้นตรง $x=b$ มีค่าเท่ากับ

\[ \rule{200pt}{1pt} \]

จงหาพื้นที่ระหว่างเส้นโค้งต่อไปนี้ พร้อมวาดกราฟโดยคร่าว

\begin{tasks}[after-skip=0pt, after-item-skip = 20pt](1)
\task เส้นตรง $y = x$, เส้นตรง $y = -x$, เส้นตรง $x=0$ และเส้นตรง $x = 6$

\hfill
\begin{tikzpicture}
\draw[<->] (-2,0) -- (2,0) node[right] {$x$};
\draw[<->] (0,-0.6) -- (0,2) node[above] {$y$};
\end{tikzpicture}

\task เส้นโค้ง $y=6-3x^2$, เส้นตรง $y=3$, ในช่วง $-1 < x < 1$

\hfill
\begin{tikzpicture}
\draw[<->] (-2,0) -- (2,0) node[right] {$x$};
\draw[<->] (0,-0.6) -- (0,2) node[above] {$y$};
\end{tikzpicture}

\task เส้นโค้ง $y=x^2+1$ และ เส้นตรง $y=x+1$, ในช่วง $x \in [0,1]$

\hfill
\begin{tikzpicture}
\draw[<->] (-2,0) -- (2,0) node[right] {$x$};
\draw[<->] (0,-0.6) -- (0,2) node[above] {$y$};
\end{tikzpicture}

\task เส้นโค้ง $y = \cos x$ และ เส้นโค้ง $y = \cos (3x)$ ระหว่างค่า $x = 0$ ถึง $x = \frac{\pi}{2}$

\hfill
\begin{tikzpicture}
\draw[<->] (-2,0) -- (2,0) node[right] {$x$};
\draw[<->] (0,-0.6) -- (0,2) node[above] {$y$};
\end{tikzpicture}

\end{tasks}

%2
\item \textbf{พื้นที่ระหว่างเส้นโค้ง ในทิศทางของแกน $y$}

ถ้า $v(y) \geq u(y)$ สำหรับทุกค่า $y$ ในช่วง $[a,b]$ แล้วพื้นที่ระหว่างเส้นโค้ง $x=u(y)$, $x=v(y)$ เส้นตรง $y =a$ และเส้นตรง $y=b$ มีค่าเท่ากับ

\[ \rule{200pt}{1pt} \]

จงหาพื้นที่ระหว่างเส้นโค้งต่อไปนี้ ในทิศทางของแกน $y$ พร้อมวาดกราฟโดยคร่าว

\begin{tasks}[after-skip=80pt, after-item-skip = 80pt](2)
\task เส้นโค้ง $x = 2-2y$, แกน $y$ และเส้นตรง $y = 6$

\hfill
\begin{tikzpicture}
\draw[<->] (-2,0) -- (2,0) node[right] {$x$};
\draw[<->] (0,-0.6) -- (0,2) node[above] {$y$};
\end{tikzpicture}

\task เส้นโค้ง $x = y^2$, เส้นตรง $x=4$ ระหว่าง $y=-2$ และ $y=2$

\hfill
\begin{tikzpicture}
\draw[<->] (-2,0) -- (2,0) node[right] {$x$};
\draw[<->] (0,-0.6) -- (0,2) node[above] {$y$};
\end{tikzpicture}

\end{tasks}
\item \textbf{ปริมาตรของรูปทรงตันที่เกิดจากการหมุนพื้นที่โดยวิธีจาน}

\begin{tasks}[after-skip=120pt, after-item-skip = 120pt](2)
\task จงหาปริมาตรที่เกิดจากการหมุนพื้นที่ซึ่งล้อมรอบด้วยเส้นโค้ง $y=4x^3, x=1$ และแกน $x$ รอบแกน $x$

\task จงหาปริมาตรที่เกิดจากการหมุนพื้นที่ซึ่งล้อมรอบด้วยเส้นโค้ง $y=\cos x, x=-\pi, x=\pi$ และแกน $x$ รอบแกน $x$
\end{tasks}

\item \textbf{ปริมาตรของรูปทรงตันที่เกิดจากการหมุนพื้นที่โดยวิธีเปลือกทรงกระบอก}

\begin{tasks}[after-skip=100pt, after-item-skip = 100pt](2)
\task จงหาปริมาตรที่เกิดจากการหมุนพื้นที่ซึ่งล้อมรอบด้วยเส้นโค้ง $y=\frac{1}{x^2}, x = 2, x=4$ และแกน $x$ รอบแกน $y$

\task จงหาปริมาตรที่เกิดจากการหมุนพื้นที่ซึ่งล้อมรอบด้วยเส้นโค้ง $y=x^2, x=0, x=1$ และแกน $x$ รอบแกน $y$
\end{tasks}

\end{enumerate}
